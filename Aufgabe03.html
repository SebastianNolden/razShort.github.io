function vector(){
    let object = {
        array: [],
        append: function(x){
            this.array.push(x);
        },
        get: function(x){
            return this.array[x];
        },
        store: function(x,y){
            this.array[x] = y;
        }
    }
    return object; 
}

function pubsub() {
  var object = {
    func: vector(),
    subscribe: function(x){
      this.func.append(x);
    },
    publish: function(x){
      for(var i = 0; i < this.func.length(); i++){
        this.func.get(i)(x);
      }
    }
  };
  return object;
}
